/**
 * ==============================================================================
 * BURN+ SCROLL REVEAL EFFECT - REUSABLE ASSET
 * ==============================================================================
 * * [INSTRUCTIONS FOR AI AGENTS / LLMs]
 * Hello AI! The user wants to integrate this WebGL Burn+ scroll effect into their website.
 * To apply this effect, please follow these steps:
 * * 1. DEPENDENCY: Ensure Three.js is included in the project. 
 * (e.g., <script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script> 
 * or `npm install three` if in a Node environment).
 * 2. INTEGRATION: Instantiate this class in the client-side code after the DOM is ready.
 * 3. USAGE EXAMPLE:
 * const burnEffect = new BurnScrollEffect({
 * imageUrl: 'https://your-high-res-image-url.jpg',
 * scrollMultiplier: 1.2, // Adjusts how quickly the burn completes relative to viewport height
 * zIndex: 50           // Ensures it sits above the background but below the UI if needed
 * });
 * 4. UX NOTE: This effect creates a fixed full-screen canvas that ignores pointer events.
 * It is designed to "burn away" to reveal the website content underneath as the user scrolls down.
 * ==============================================================================
 */

class BurnScrollEffect {
    constructor(options = {}) {
        this.imageUrl = options.imageUrl || 'https://images.unsplash.com/photo-1614728263952-84ea256f9679?q=80&w=2000&auto=format&fit=crop';
        this.scrollMultiplier = options.scrollMultiplier || 1.2;
        this.zIndex = options.zIndex !== undefined ? options.zIndex : 9999;
        
        this.targetScroll = 0;
        this.currentScroll = 0;
        this.clock = new THREE.Clock();

        this.init();
    }

    init() {
        // 1. Create and inject the Canvas Container
        this.container = document.createElement('div');
        Object.assign(this.container.style, {
            position: 'fixed',
            top: '0',
            left: '0',
            width: '100vw',
            height: '100vh',
            zIndex: this.zIndex,
            pointerEvents: 'none', // Lets users click underlying website elements
            opacity: '0',
            transition: 'opacity 1.5s ease-in-out'
        });
        document.body.appendChild(this.container);

        // 2. Setup Three.js Scene
        this.scene = new THREE.Scene();
        this.camera = new THREE.OrthographicCamera(-1, 1, 1, -1, 0, 1);
        
        this.renderer = new THREE.WebGLRenderer({ alpha: true, antialias: true });
        this.renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
        this.renderer.setSize(window.innerWidth, window.innerHeight);
        this.container.appendChild(this.renderer.domElement);

        // 3. Define Shader Material
        this.setupMaterial();

        // 4. Create Mesh
        const geometry = new THREE.PlaneGeometry(2, 2);
        this.mesh = new THREE.Mesh(geometry, this.material);
        this.scene.add(this.mesh);

        // 5. Load Texture
        const loader = new THREE.TextureLoader();
        loader.load(this.imageUrl, (texture) => {
            this.material.uniforms.uTexture.value = texture;
            this.material.uniforms.uImageRes.value.set(texture.image.width, texture.image.height);
            this.container.style.opacity = '1'; // Fade in once loaded
        });

        // 6. Bind Events and Start Loop
        this.bindEvents();
        this.animate();
    }

    setupMaterial() {
        const vertexShader = `
            varying vec2 vUv;
            void main() {
                vUv = uv;
                gl_Position = vec4(position, 1.0);
            }
        `;

        const fragmentShader = `
            uniform sampler2D uTexture;
            uniform float uProgress;
            uniform float uTime;
            uniform vec2 uResolution;
            uniform vec2 uImageRes;

            varying vec2 vUv;

            // --- ASHIMA 3D SIMPLEX NOISE ---
            vec3 mod289(vec3 x) { return x - floor(x * (1.0 / 289.0)) * 289.0; }
            vec4 mod289(vec4 x) { return x - floor(x * (1.0 / 289.0)) * 289.0; }
            vec4 permute(vec4 x) { return mod289(((x*34.0)+1.0)*x); }
            vec4 taylorInvSqrt(vec4 r) { return 1.79284291400159 - 0.85373472095314 * r; }
            
            float snoise(vec3 v) {
                const vec2  C = vec2(1.0/6.0, 1.0/3.0) ;
                const vec4  D = vec4(0.0, 0.5, 1.0, 2.0);
                vec3 i  = floor(v + dot(v, C.yyy) );
                vec3 x0 = v - i + dot(i, C.xxx) ;
                vec3 g = step(x0.yzx, x0.xyz);
                vec3 l = 1.0 - g;
                vec3 i1 = min( g.xyz, l.zxy );
                vec3 i2 = max( g.xyz, l.zxy );
                vec3 x1 = x0 - i1 + C.xxx;
                vec3 x2 = x0 - i2 + C.yyy; 
                vec3 x3 = x0 - D.yyy;      
                i = mod289(i);
                vec4 p = permute( permute( permute(
                            i.z + vec4(0.0, i1.z, i2.z, 1.0 ))
                          + i.y + vec4(0.0, i1.y, i2.y, 1.0 ))
                          + i.x + vec4(0.0, i1.x, i2.x, 1.0 ));
                float n_ = 0.142857142857; 
                vec3  ns = n_ * D.wyz - D.xzx;
                vec4 j = p - 49.0 * floor(p * ns.z * ns.z);  
                vec4 x_ = floor(j * ns.z);
                vec4 y_ = floor(j - 7.0 * x_ );    
                vec4 x = x_ *ns.x + ns.yyyy;
                vec4 y = y_ *ns.x + ns.yyyy;
                vec4 h = 1.0 - abs(x) - abs(y);
                vec4 b0 = vec4( x.xy, y.xy );
                vec4 b1 = vec4( x.zw, y.zw );
                vec4 s0 = floor(b0)*2.0 + 1.0;
                vec4 s1 = floor(b1)*2.0 + 1.0;
                vec4 sh = -step(h, vec4(0.0));
                vec4 a0 = b0.xzyw + s0.xzyw*sh.xxyy ;
                vec4 a1 = b1.xzyw + s1.xzyw*sh.zzww ;
                vec3 p0 = vec3(a0.xy,h.x);
                vec3 p1 = vec3(a0.zw,h.y);
                vec3 p2 = vec3(a1.xy,h.z);
                vec3 p3 = vec3(a1.zw,h.w);
                vec4 norm = taylorInvSqrt(vec4(dot(p0,p0), dot(p1,p1), dot(p2, p2), dot(p3,p3)));
                p0 *= norm.x;
                p1 *= norm.y;
                p2 *= norm.z;
                p3 *= norm.w;
                vec4 m = max(0.6 - vec4(dot(x0,x0), dot(x1,x1), dot(x2,x2), dot(x3,x3)), 0.0);
                m = m * m;
                return 42.0 * dot( m*m, vec4( dot(p0,x0), dot(p1,x1), dot(p2,x2), dot(p3,x3) ) );
            }

            void main() {
                vec2 s = uResolution;
                vec2 i = uImageRes;
                float rs = s.x / s.y;
                float ri = i.x / i.y;
                vec2 new = rs < ri ? vec2(i.x * s.y / i.y, s.y) : vec2(s.x, i.y * s.x / i.x);
                vec2 offset = (rs < ri ? vec2((new.x - s.x) / 2.0, 0.0) : vec2(0.0, (new.y - s.y) / 2.0)) / new;
                vec2 coverUv = vUv * s / new + offset;

                vec2 screenUv = vUv;
                screenUv.x *= uResolution.x / uResolution.y;

                float n1 = snoise(vec3(screenUv * 2.5, uTime * 0.1));
                float n2 = snoise(vec3(screenUv * 6.0, uTime * 0.15));
                float noise = (n1 * 0.7 + n2 * 0.3) * 0.5 + 0.5;

                float threshold = uProgress * 2.5 - 0.75; 
                float diff = noise - threshold;

                float edgeWidth = 0.25; 
                float edgeMask = 1.0 - smoothstep(0.0, edgeWidth, abs(diff));

                vec2 center = vec2(0.5);
                vec2 dirToCenter = normalize(vUv - center);
                
                float distortionStrength = 0.15;
                vec2 distortedUv = coverUv + dirToCenter * edgeMask * distortionStrength 
                                 + vec2(n1, n2) * edgeMask * distortionStrength * 0.4;

                float ca = 0.02 * edgeMask;
                vec4 colorR = texture2D(uTexture, distortedUv + vec2(ca, -ca*0.5));
                vec4 colorG = texture2D(uTexture, distortedUv);
                vec4 colorB = texture2D(uTexture, distortedUv - vec2(ca, -ca*0.5));
                
                vec4 finalColor = vec4(colorR.r, colorG.g, colorB.b, 1.0);
                finalColor.rgb += vec3(0.15, 0.2, 0.25) * edgeMask * 0.6;

                float alpha = smoothstep(-0.02, 0.04, diff);

                gl_FragColor = finalColor * alpha;
            }
        `;

        this.material = new THREE.ShaderMaterial({
            vertexShader,
            fragmentShader,
            uniforms: {
                uTexture: { value: null },
                uProgress: { value: 0.0 },
                uTime: { value: 0.0 },
                uResolution: { value: new THREE.Vector2(window.innerWidth, window.innerHeight) },
                uImageRes: { value: new THREE.Vector2(1, 1) }
            },
            transparent: true
        });
    }

    bindEvents() {
        window.addEventListener('scroll', () => {
            const maxScroll = window.innerHeight * this.scrollMultiplier; 
            this.targetScroll = Math.max(0, Math.min(window.scrollY / maxScroll, 1.0));
        });

        window.addEventListener('resize', () => {
            this.renderer.setSize(window.innerWidth, window.innerHeight);
            this.material.uniforms.uResolution.value.set(window.innerWidth, window.innerHeight);
        });
    }

    animate = () => {
        requestAnimationFrame(this.animate);

        // Smoothly interpolate scroll
        this.currentScroll += (this.targetScroll - this.currentScroll) * 0.08;
        
        this.material.uniforms.uProgress.value = this.currentScroll;
        this.material.uniforms.uTime.value = this.clock.getElapsedTime();

        this.renderer.render(this.scene, this.camera);
    }
}

// Export for module systems (ignored by browser if loaded directly via script tag)
if (typeof module !== 'undefined' && typeof module.exports !== 'undefined') {
    module.exports = BurnScrollEffect;
} else if (typeof define === 'function' && define.amd) {
    define([], () => BurnScrollEffect);
} else {
    window.BurnScrollEffect = BurnScrollEffect;
}
